WHERE {
  <http://www.wikidata.org/entity/Q7604>  <http://www.wikidata.org/prop/direct/P802> + ?var1 .
  ?var1  ?var2  ?var3 .
 FILTER (  ( (  ?var2  =  <http://www.wikidata.org/prop/direct/P1066>  ) ) 
) .
 SERVICE  <http://wikiba.se/ontology#label>   {
    <http://www.bigdata.com/rdf#serviceParam>  <http://wikiba.se/ontology#language>  "en,en".
  }
